package release_build_secretscan

# The secretscan attestor (gitleaks under the hood) scans the entire
# product set + every prior attestation envelope for matching secret
# patterns, with multi-layer base64/hex/URL decode passes. When it
# finds something it replaces the actual value with a SHA-256 digest
# of the secret, so the predicate can be safely stored.
#
# Filtering: SBOM files (.spdx.json, .cyclonedx.json) legitimately
# contain long hex strings — Go module commit SHAs, file digests,
# license IDs — that match several gitleaks token regexes (notably
# sourcegraph-access-token, which is "anything 40 hex chars"). These
# are NOT secrets; they're public identifiers. Allowlist them by
# (ruleId, location) pair so a real high-entropy finding in the
# binary itself still trips the gate.
#
# Any finding that is NOT a known-false-positive class fails the
# release. The allowlist is intentionally narrow — `is_known_fp`
# only matches patterns we've audited; everything else is treated
# as a real finding.

is_known_fp(f) {
  endswith(f.location, ".spdx.json")
  f.ruleId == "sourcegraph-access-token"
}

is_known_fp(f) {
  endswith(f.location, ".cyclonedx.json")
  f.ruleId == "sourcegraph-access-token"
}

# Gitleaks's `sourcegraph-access-token` rule is `\b[a-f0-9]{40}\b` — it
# matches ANY 40-char hex string. The git attestor's predicate
# legitimately carries multiple 40-char SHAs (commithash, parenthashes,
# treehash) that all match. None of these are secrets — they're public
# git identifiers. Suppress for the entire `attestation:git*` location
# prefix.
is_known_fp(f) {
  startswith(f.location, "attestation:git")
  f.ruleId == "sourcegraph-access-token"
}

# Same shape risk in the github attestor (workflow URLs contain SHAs
# in some cases) and material attestor (file digests).
is_known_fp(f) {
  startswith(f.location, "attestation:github")
  f.ruleId == "sourcegraph-access-token"
}

is_known_fp(f) {
  startswith(f.location, "attestation:material")
  f.ruleId == "sourcegraph-access-token"
}

real_findings := [f |
  f := input.findings[_]
  not is_known_fp(f)
]

deny[msg] {
  count(real_findings) > 0
  n := count(real_findings)
  msg := sprintf("secretscan: %d genuine finding(s) after FP filter — release blocked", [n])
}
