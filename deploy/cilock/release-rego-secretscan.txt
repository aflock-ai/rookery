package release_build_secretscan

# The secretscan attestor (gitleaks under the hood) scans the entire
# product set + every prior attestation envelope for matching secret
# patterns, with multi-layer base64/hex/URL decode passes. When it
# finds something it replaces the actual value with a SHA-256 digest
# of the secret, so the predicate can be safely stored — but a
# non-empty findings array means we shipped potential credentials
# into a publicly-released artifact, which is the highest-severity
# release defect we can have.
#
# Zero tolerance: ANY finding fails the release.

deny[msg] {
  count(input.findings) > 0
  n := count(input.findings)
  msg := sprintf("secretscan found %d potential secret(s) — release blocked", [n])
}
