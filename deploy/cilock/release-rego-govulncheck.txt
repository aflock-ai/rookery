package release_build_govulncheck

# govulncheck runs source-mode against the cilock module and reports
# OSV findings whose vulnerable SYMBOLS are reachable from main(). The
# attestor parses govulncheck's streaming-JSON wire format into a typed
# Summary predicate. Fields we gate on:
#
#   summary.reachableCount  — count of OSV findings reachable from main
#   summary.bySeverity      — severity buckets (critical/high/medium/low/unknown)
#
# Reachable findings are the actionable ones: govulncheck has already
# proven the vulnerable function is in the binary's call graph. We deny
# on ANY reachable finding regardless of severity, because Go vuln DB
# entries don't currently carry CVSS so almost everything is in the
# "unknown" bucket — relying on bySeverity would be a no-op.
#
# Note on cross-compilation: source-mode runs against the source tree,
# not the cross-compiled binary, so this gate applies the same set of
# findings to all platform legs (the linked symbol set is the same).
# We accept this — if there's a reachable CVE in linux/amd64, the same
# CVE is reachable in darwin/arm64.

deny[msg] {
  input.summary.reachableCount > 0
  msg := sprintf("govulncheck: %d reachable vulnerability finding(s) — release blocked", [input.summary.reachableCount])
}

# Belt-and-suspenders: also deny if any critical/high severity finding
# exists at all (reachable or not). Reachable count is the primary gate,
# but a high-severity unreachable today can become reachable with a
# refactor tomorrow — surfacing it now lets us decide.

deny[msg] {
  input.summary.bySeverity.critical > 0
  msg := sprintf("govulncheck: %d CRITICAL severity finding(s) — release blocked", [input.summary.bySeverity.critical])
}

deny[msg] {
  input.summary.bySeverity.high > 0
  msg := sprintf("govulncheck: %d HIGH severity finding(s) — release blocked", [input.summary.bySeverity.high])
}
