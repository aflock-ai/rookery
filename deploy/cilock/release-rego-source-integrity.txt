package release_source_integrity

# Source must be in a clean, identifiable state. If the working tree has
# any modified, staged, or untracked files at build time, the artifact
# the build produced does not correspond to a known git commit — could
# be a build script that wrote into the source tree, a post-clone
# patch, or a code-injection step.

deny[msg] {
  count(input.status) > 0
  changed := [path | input.status[path]]
  msg := sprintf("source tree dirty at build time — %d file(s) had changes: %v", [count(input.status), changed])
}

# The commit hash must be present and non-empty. A missing commit hash
# means git couldn't be queried — either not a repo, detached HEAD, or
# the .git directory was stripped before the build. Refuse to gate on
# an unrooted source tree.
deny[msg] {
  not input.commithash
  msg := "git commit hash missing — source state cannot be verified"
}

deny[msg] {
  input.commithash == ""
  msg := "git commit hash is empty"
}
