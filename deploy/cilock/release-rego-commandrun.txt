package release_build_commandrun

# The release build must use the canonical hardening flags:
#   - CGO_ENABLED=0 (no C compiler linking — eliminates dynamic library injection)
#   - -trimpath      (strips local paths from binary; reproducible)
#   - -ldflags ... cli.Version=v...  (version is embedded, not "dev")
#
# Without these the binary doesn't match what install docs promise and
# auditors can't verify the build is reproducible from source.

cmd := concat(" ", input.cmd)

deny[msg] {
  not contains(cmd, "-trimpath")
  msg := "release build missing -trimpath flag"
}

deny[msg] {
  not contains(cmd, "cli.Version=")
  msg := "release build missing -ldflags Version embed"
}

deny[msg] {
  some env in input.environment.variables
  env.name == "CGO_ENABLED"
  env.value != "0"
  msg := sprintf("CGO_ENABLED must be 0 in release builds, got: %s", [env.value])
}
