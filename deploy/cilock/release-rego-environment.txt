package release_build_environment

# CGO_ENABLED=0 is required for release builds. CGO linking pulls in
# the system C compiler/loader and dynamically links against system
# libraries — that breaks the "static, no surprise dependencies"
# guarantee end users rely on. Pure-Go binaries also dodge the entire
# class of glibc-version compatibility footguns.

deny[msg] {
  input.variables.CGO_ENABLED != "0"
  msg := sprintf("CGO_ENABLED must be 0 in release builds, got: %s", [input.variables.CGO_ENABLED])
}

# GOFIPS140 selects the FIPS 140-3 Go Cryptographic Module version to
# compile in (Go 1.24+). Without this set, the binary cannot operate in
# FIPS mode at runtime even with //go:debug fips140=on — the FIPS
# module simply isn't present. We require a non-empty, non-"off" value.
# FedRAMP 20x and NIST 800-204D ratepayers will check this.

deny[msg] {
  not input.variables.GOFIPS140
  msg := "GOFIPS140 unset — FIPS 140-3 Go Cryptographic Module not compiled into release build"
}

deny[msg] {
  input.variables.GOFIPS140 == ""
  msg := "GOFIPS140 empty — FIPS 140-3 Go Cryptographic Module not compiled into release build"
}

deny[msg] {
  input.variables.GOFIPS140 == "off"
  msg := "GOFIPS140=off — FIPS 140-3 Go Cryptographic Module explicitly disabled in release build"
}
